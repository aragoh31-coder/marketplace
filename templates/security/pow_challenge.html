Manual Solver (No JS):

# Python example:
import hashlib

challenge = "{{ challenge.challenge.pow_data.challenge }}"
difficulty = {{ difficulty }}
target = "0" * difficulty

nonce = 0
while True:
    solution = f"{challenge}:{nonce}"
    hash_result = hashlib.sha256(solution.encode()).hexdigest()
    if hash_result.startswith(target):
        print(f"Found solution! Nonce: {nonce}")
        print(f"Hash: {hash_result}")
        break
    nonce += 1
